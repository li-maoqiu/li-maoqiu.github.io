〈h5|获取当前已缓存的大小〉
let size=0;
for(item in window.localStorage){
    if(window.localStorage.hasOwnProperty(item)){
        size+=window.localStorage.getItem(item).length;
    }
}
console.log((size/1024).toFixed(2)+'KB')
